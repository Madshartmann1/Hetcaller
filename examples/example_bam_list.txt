# Example BAM list file for processing multiple samples
# Format: /path/to/file.bam [optional_output_prefix]
# Lines starting with # are ignored
# If no prefix is provided, the BAM filename (without .bam) is used

# Example with explicit prefixes:
/data/samples/sample1.bam sample1
/data/samples/sample2.bam sample2
/data/samples/sample3.bam sample3

# Example without prefixes (will use filenames):
# /data/samples/bear_001.bam
# /data/samples/bear_002.bam
# /data/samples/elephant_001.bam
